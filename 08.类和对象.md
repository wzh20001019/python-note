#### 类和对象

```python
class Student:
    # 构造方法 __init__(self)
    def __init__(self, name, age):
        self.name = name
        self.age = age

    # def __eq__(self, o: object) -> bool:
    #     return super().__eq__(o)

    # 类似 java toString
    def __str__(self):
        return f"Student, name:{self.name}, age:{self.age}"


stu = Student('wzh', 20)

print(stu)
```
